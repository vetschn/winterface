156  private:
158  virtual int sync() noexcept {
159  const int r1 = sb1_->pubsync();

162  }
164  virtual int_type overflow(const int_type c) noexcept {
165  const int_type eof = TT::eof();
166  if (TT::eq_int_type(c, eof)) return TT::not_eof(c);
167 
168  const char_type ch = TT::to_char_type(c);
169  const int_type r1 = sb1_->sputc(ch);
170  const int_type r2 = sb2_->sputc(ch);
171 
172  return TT::eq_int_type(r1, eof) || TT::eq_int_type(r2, eof) ? eof : c;
173  }

175  private:
