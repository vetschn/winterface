174 inline CT parseFile(const std::string &fileName, size_t &lcnt, size_t &kcnt,
175  std::ostream &os = std::cout) {
176  // open file
177  auto file = aux::openFile<std::ifstream>(fileName);
178 
179  // result container
180  CT res;
181 
182  // set of parsed keys
183  std::set<uint32_t> pk;
184 
185  // regex
186  const std::regex rgx(RGX__);
187 
188  // read line by line and tokenize
189  lcnt = 0; // line counter
190  try {
191  std::string line;
192  while (std::getline(file, line)) {
193  ++lcnt;
194 
195  // resize line to first occurrence of comment symbol
196  line = line.substr(0, line.find_first_of(ICOM__));
197 
198  std::sregex_token_iterator i(line.begin(), line.end(), rgx, -1), e;
199  if (i == e) continue;
200  if (!i->length()) ++i; // line starts with delimiter
201  if (i == e) continue;
202 
203  // hash key and check for duplicate
204  const uint32_t key = aux::fnvHash(i->str().c_str());
205  if (pk.find(key) != pk.end())
206  throw(std::invalid_argument("duplicate key \'" + i->str() + "\'"));
207 
208  // parse key
209  typename CT::parseKey_ prsk(res, key, file, lcnt, i, e);
210  if (!prsk.rcnt())
211  throw(std::invalid_argument("invalid key \'" + i->str() + "\'"));
212 
213  // insert parsed key
214  pk.insert(key);
215  }
216  } catch (const std::exception &e) {
217  throw(std::invalid_argument("file: " + fileName + ", line " +
218  std::to_string(lcnt) + ", " + e.what()));
219  }
220 
221  // check whole file was parsed and close file
222  assert(file.eof());
223  file.close();
224 
225  kcnt = pk.size(); // key counter
226  if (res.verbosity) {
227  os << "file \'" << fileName << "\' successfully parsed\n";
228  os << kcnt << " keys parsed on " << lcnt << " lines\n\n";
229  }
230 
231  return res;
232 }

522  const size_t n, std::ifstream &file, size_t &lcnt,
523  std::sregex_token_iterator &i,
524  const std::sregex_token_iterator &e) {
525  bitch_(i, e);

527 
528  inp.clear();
529  auto extRes = [&inp](const std::string &tok) -> void {
530  const size_t mpos = tok.find_first_of("-:");
531 
532  if (!mpos) throw(std::invalid_argument("indices must be >=0"));
533  if (mpos > tok.size()) // one number
534  inp.push_back(std::stoul(tok));
535  else { // range
536  size_t l = std::stoul(tok.substr(0, mpos));
537  size_t u = std::stoul(tok.substr(mpos + 1));
538  if (u <= l) std::swap(l, u);
539 
540  inp.reserve(inp.size() + u - l);
541  for (size_t j = l; j <= u; ++j) inp.push_back(j);
542  }
543  };
544  if (n < NPOS__) {
545  inp.reserve(n);
546  while (i != e && inp.size() <= n) extRes((i++)->str());
547  } else {
548  inp.reserve(ALLOC__);
549  while (i != e) {
550  if (inp.size() == inp.capacity()) inp.reserve(inp.size() + ALLOC__);
551  extRes((i++)->str());
552  }
