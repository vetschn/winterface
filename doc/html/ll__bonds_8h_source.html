677  inline ll_cell getBondCenters() const noexcept {
678  if (this->empty()) return ll_cell(cell().B());
679 
680  fMat bc(dim(), 0);
681  bc.reserve(this->cardinality() / 2);
682  ll__::idv bid;
683  bid.reserve(bc.ccap());
684  const std::regex r_("[_]+");
685 
686  // find centers
687  for (auto i = this->cbegin(), ie = this->cend(); i != ie; ++i) {
688  if (i->i1() > i->i2()) continue; // skip inverted partners
689  const auto cpbc = cell().cAt(i->i1()) + this->bond(*i) * .5;
690 
691  // if i1==i2 inverted partners are among the 2nd half of R vectors
692  for (auto r = i->ccBegin(),
693  re = (i->i1() == i->i2() ? i->ccBegin() + i->N() / 2
694  : i->ccEnd());
695  r != re; ++r) {
696  const auto cbc = (cpbc + *r * .5) % 1.0;
697  const auto itr = std::lower_bound(bc.ccBegin(), bc.ccEnd(), cbc);
698  if (itr != bc.ccEnd() && *itr == cbc) {
699  // center already included, dump all to buff vector
700  std::vector<std::string> buff;
701  for (std::sregex_token_iterator i(bid[(size_t)itr].cbegin(),
702  bid[(size_t)itr].cend(), r_, -1),
703  e;
704  i != e; ++i)
705  buff.push_back(*i);
706 
707  const std::string id1 = cell().id(cell().type(i->i1()));
708  const std::string id2 = cell().id(cell().type(i->i2()));
709  buff.push_back(id1 < id2 ? '(' + id1 + ':' + id2 + ')'
710  : '(' + id2 + ':' + id1 + ')');
711 
712  // sort, remove duplicates and reconstruct
713  std::sort(buff.begin(), buff.end());
714  buff.resize(
715  distance(buff.begin(), std::unique(buff.begin(), buff.end())));
716  bid[(size_t)itr] = "";
717  for (const auto &s : buff) bid[(size_t)itr] += (s + "_");
718  bid[(size_t)itr].pop_back();
719 
720  } else {
721  // new center
722  bc.cInsert(itr, cbc);
723 
724  const std::string id1 = cell().id(cell().type(i->i1()));
725  const std::string id2 = cell().id(cell().type(i->i2()));
726  bid.insert(bid.cbegin() + (size_t)itr,
727  id1 < id2 ? '(' + id1 + ':' + id2 + ')'
728  : '(' + id2 + ':' + id1 + ')');
729  }
730  }
731  }
732 
733  // add parenthesis around ids with '_' in them
734  for (auto &s : bid)
735  if (s.find('_') != std::string::npos) s = '(' + s + ')';
736 
737  // return
738  return ll_cell(cell().B(), std::move(bc), std::move(bid));
739  }
