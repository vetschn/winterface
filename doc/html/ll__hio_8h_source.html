272  inline ll_writerW90(const std::string &fileName, lm__::fMat R,
273  const size_t Nw, const size_t prec = 12)
274  : ll_writer(std::move(R)),
275  fileName_(fileName),
276  file_(aux::openFile<std::ofstream>(fileName)),
277  prec_(prec) {
278  // write header
279  {
280  std::locale::global(std::locale("en_US.utf8"));
281  std::time_t t = std::time(nullptr);
282  char mbstr[100];
283  std::strftime(mbstr, sizeof(mbstr), "%c", std::localtime(&t));
284 
285  file_ << mbstr << "\n";
286  }
287 
288  // write Nw, Nr
289  file_ << std::setw(precision()) << Nw << "\n";
290  file_ << std::setw(precision()) << NR() << "\n";
291 
292  // fake degeneracy
293  for (size_t i = 1; i <= NR(); ++i)
294  file_ << std::setw(5) << 1 << (!(i % 15) && i != NR() ? "\n" : "");
295  file_ << "\n";
296  }

303  inline const std::string fileName() const noexcept { return fileName_; }
304 
307  inline std::string c_descr() const noexcept {
309  return lm__::T(this->c_R()).print(0);
310  }
311 
314  inline void newBlock() {}
316 
317  private:
320  inline size_t bytes_() const noexcept {
322  return 2 * sizeof(size_t) + sizeof(ll__::hel);
323  }
327  inline void insert_(const hdat &inp) {
328  for (const auto &h : inp) {
329  for (const int i : c_R()) file_ << std::setw(8) << i;
330  file_ << std::setw(8) << h.m << std::setw(8) << h.n
331  << std::setw(precision() + 8) << std::fixed
332  << std::setprecision(precision()) << std::real(h.h)
333  << std::setw(precision() + 8) << std::fixed
334  << std::setprecision(precision()) << std::imag(h.h) << "\n";
335  }
