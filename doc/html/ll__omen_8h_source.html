286 inline std::string H_fileName(const lm__::fArray &R) noexcept {
287  assert(R.size() == DIM__);
288  return "H_" +
289  std::to_string((DIM__ * DIM__) / 2 +
290  int(R[0] * DIM__ * DIM__ + R[1] * DIM__ + R[2])) +
291  ".bin";
292 }
294 inline fMat rToR(const rv &r) noexcept {
295  assert(r.size() == DIM__);
296 
297  // get bounds and total number of shift vectors
298  fMat bnds(DIM__, 1);
299  for (size_t d = 0; d != DIM__; ++d) bnds[d] = r[d] ? 0.0 : 1.0;
300 
301  // result, allocate space
302  fMat res(DIM__, 0);
303  res.reserve(size_t(std::pow(DIM__, sum(bnds))));
304 
305  // generate R vecs
306  fMat vec(DIM__, 1);
307  for (vec[0] = -bnds[0]; vec[0] <= bnds[0]; ++vec[0])
308  for (vec[1] = -bnds[1]; vec[1] <= bnds[1]; ++vec[1])
309  for (vec[2] = -bnds[2]; vec[2] <= bnds[2]; ++vec[2]) res.push_back(vec);
310 
311  return res;
312 }
