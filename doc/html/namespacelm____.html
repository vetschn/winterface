Initial value:
= [](const auto& v1, const auto& v2) -> bool {
assert(v1.size() == v2.size());
auto j = v2.cbegin();
for (auto i = v1.cbegin(), e = v1.cend(); i != e; ++i, ++j) {
if (ops::lt(*i, *j)) return true;
if (ops::gt(*i, *j)) return false;
}
return false;
